\section{Motivation}
\begin{frame}{Motivation \& Central Question}
\begin{block}{Central Question}
  \centering\textit{Under what conditions does boosting outperform bagging,\\
  and vice versa --- and \textbf{why}?}
\end{block}

\bigskip
\begin{columns}
\column{0.5\textwidth}
\textbf{Boosting (AdaBoost / SAMME)}
\begin{itemize}
  \item Re-weights hard examples sequentially
  \item Drives down bias
  \item Sensitive to label noise
\end{itemize}
\column{0.5\textwidth}
\textbf{Bagging (Random Forest)}
\begin{itemize}
  \item Bootstrap aggregation, trained independently
  \item Drives down variance
  \item More robust to noise
\end{itemize}
\end{columns}

\bigskip
\begin{itemize}
  \item All algorithms implemented from scratch in NumPy (no \texttt{sklearn} ensemble classes)
  \item Verified against \texttt{sklearn} on every dataset; 190 tests, 95\% line coverage
\end{itemize}
\end{frame}

\section{Methods}
\begin{frame}{Module 1 -- Decision Tree (CART)}
\begin{itemize}
  \item Binary CART classifier: Gini impurity or entropy splitting
  \item Supports weighted samples (required by AdaBoost)
  \item \texttt{max\_features} subsampling (required by Random Forest)
  \item Feature importances; matches sklearn within 2\% on all four datasets
\end{itemize}
\vfill
\[
  I_G(\mathbf{p}) = 1 - \sum_{c=1}^{C} p_c^2
  \qquad\qquad
  \Delta I = I_{\text{parent}} - \frac{N_L}{N}I_L - \frac{N_R}{N}I_R
\]
\end{frame}

% ------------------------------------------------------------------
\begin{frame}{Module 2 -- AdaBoost (SAMME)}
Discrete SAMME using depth-1 decision stumps as weak learners. For $m = 1 \ldots M$:
\begin{enumerate}
  \item Fit stump $h_m$ with sample weights $\mathbf{w}^{(m)}$
  \item $\epsilon_m = \sum_i w_i^{(m)}\, \mathbf{1}[h_m(x_i)\neq y_i]$
  \item $\alpha_m = \ln\dfrac{1-\epsilon_m}{\epsilon_m} + \ln(K-1)$
  \item $w_i^{(m+1)} \propto w_i^{(m)} \exp\!\bigl(\alpha_m \mathbf{1}[h_m(x_i)\neq y_i]\bigr)$, renormalise
\end{enumerate}
\vfill
Final prediction: $\hat{y} = \arg\max_k \sum_{m=1}^{M} \alpha_m \mathbf{1}[h_m(x)=k]$

\medskip
Bonus multi-class extensions: one-vs-rest wrapper and SAMME.R (real-valued).
\end{frame}

% ------------------------------------------------------------------
\begin{frame}{Module 3 -- Random Forest}
\begin{itemize}
  \item $T$ trees, each on a bootstrap sample $\mathcal{D}_t$ (drawn with replacement)
  \item Each split considers only $\lfloor\sqrt{p}\rfloor$ random features
  \item Prediction by majority vote; \texttt{predict\_proba} averages per-tree probabilities
  \item Out-of-bag (OOB) scoring: no-extra-cost validation proxy
  \item Parallel training via \texttt{multiprocessing.Pool}
\end{itemize}
\end{frame}

\section{Discussion \& Conclusion}
\begin{frame}{Key Findings}
\begin{block}{When does Boosting win?}
  Binary, reasonably separable classes, clean labels -- lower bias$^2$, best accuracy on
  Breast Cancer.
\end{block}
\begin{block}{When does Bagging win?}
  Severe class imbalance, high dimensionality, or $>$2 classes -- lower variance,
  native multi-class handling, more consistent generalisation.
\end{block}
\begin{block}{Unsupervised insight}
  High K-Means ARI (Breast Cancer) $\leftrightarrow$ best supervised accuracy;
  near-zero ARI (Credit Card, MNIST) $\leftrightarrow$ where RF's conservative variance
  profile earns its keep.
\end{block}
\end{frame}
